\section{$t_8 t_8 RR|DP|^2$}
One realizes that \eqref{eq:t8R2S2} in fact also holds for the cas $RRS\bar{S}$
\begin{equation}
\begin{aligned}
t_8 t_8 R^2 S \bar{S} &=
{}4\,R^{m n r s} R_{m n r s} S^{t u v w} \bar{S}\,_{t u v w}+16\,R^{m n r s} R_{m n}\,^{t u} S_{r s}\,^{v w} \bar{S}\,_{t u v w}\\
&-64\,R^{m n r s} R_{m n r}\,^{t} S_{s}\,^{u v w} \bar{S}\,_{t u v w}-32\,R^{m n r s} R_{m n}\,^{t u} S_{r t}\,^{v w} \bar{S}\,_{s u v w}\\
&+8\,R^{m n r s} R^{t u v w} S_{m n r s} \bar{S}\,_{t u v w}+8\,R^{m n r s} R^{t u v w} S_{m n t u} \bar{S}\,_{r s v w}\\
&-64\,R^{m n r s} R_{m}\,^{t u v} S_{n}\,^{w}\,_{u v} \bar{S}\,_{r s t w}-64\,R^{m n r s} R_{m}\,^{t u v} S_{n}\,^{w}\,_{r s} \bar{S}\,_{t w u v}\\
&-64\,R^{m n r s} R^{t u v w} S_{m n r t} \bar{S}\,_{s u v w}+128\,R^{m n r s} R_{m}\,^{t}\,_{r}\,^{u} S_{n}\,^{v}\,_{s}\,^{w} \bar{S}\,_{t v u w}\\
&+128\,R^{m n r s} R_{m}\,^{t}\,_{r}\,^{u} S_{n}\,^{v}\,_{u}\,^{w} \bar{S}\,_{s w t v}+256\,R^{m n r s} R_{m}\,^{t u v} S_{n}\,^{w}\,_{r u} \bar{S}\,_{s v t w}\\
&+64\,R^{m n r s} R^{t u v w} S_{m t r v} \bar{S}\,_{n u s w}
\end{aligned}
\end{equation}
Letting $S_{mnrs} = DP_{mnrs}$, and substituting the eqm $D_m P^m = 0$, as well as $R_{mn} = -2 \bar{P}_{(m} P_{n)}$, we find that the above expression reduces to
\begin{equation}
\begin{aligned}
t_8 t_8 R^2 |DP^2|
&=
{}32\,D_{m} P_{n}\, R_{r}\,^{s}\,_{t}\,^{u} R^{r m t n}\, D_{s} \bar{P}_{u}
+96\,D_{m} P_{n}\, R_{r}\,^{n}\,_{s}\,^{t} R^{r m s u}\, D_{u} \bar{P}_{t}\\
&-16\,D_{m} P_{n}\, R_{r s}\,^{n t} R^{r s m u}\, D_{u} \bar{P}_{t}
-32\,D_{m} P^{n}\, R_{r s t}\,^{u} R^{r s t m}\, D_{u} \bar{P}_{n}\\
&+4\,D^{m} P^{n}\, R_{r s t u} R^{r s t u}\, D_{m} \bar{P}_{n}
-32\,D_{m} P_{n}\, P^{m} R_{r}\,^{s n t} \bar{P}\,^{r}\, D_{s} \bar{P}_{t}\\
&-32\,D_{m} P_{n}\, P^{r} R_{r}\,^{s n t} \bar{P}\,^{m}\, D_{s} \bar{P}_{t}
+16\,D_{m} P_{n}\, P^{r} R^{m s n t} \bar{P}_{r}\, D_{s} \bar{P}_{t}\\
&+32\,D_{m} P_{n}\, P^{r} R_{r}\,^{m n s} \bar{P}\,^{t}\, D_{t} \bar{P}_{s}
+32\,D_{m} P_{n}\, P^{r} R_{s}\,^{m n t} \bar{P}\,^{s}\, D_{r} \bar{P}_{t}\\
&-32\,D_{m} P^{n}\, P^{r} R_{r}\,^{m}\,_{s}\,^{t} \bar{P}\,^{s}\, D_{t} \bar{P}_{n}
-32\,D_{m} P^{n}\, P^{r} R_{r}\,^{s}\,_{t}\,^{m} \bar{P}\,^{t}\, D_{s} \bar{P}_{n}\\
&+32\,D_{m} P_{n}\, P^{m} P^{r} \bar{P}\,^{n} \bar{P}\,^{s}\, D_{r} \bar{P}_{s}
-16\,D_{m} P_{n}\, P^{m} P^{n} \bar{P}\,^{r} \bar{P}\,^{s}\, D_{r} \bar{P}_{s}\\
&-16\,D_{m} P_{n}\, P^{r} P^{s} \bar{P}\,^{m} \bar{P}\,^{n}\, D_{r} \bar{P}_{s}
+16\,D^{m} P^{n}\, P_{r} P^{r} \bar{P}_{s} \bar{P}\,^{s}\, D_{m} \bar{P}_{n}\\
&+16\,D^{m} P^{n}\, P^{r} P^{s} \bar{P}_{r} \bar{P}_{s}\, D_{m} \bar{P}_{n}
\end{aligned}
\end{equation}
