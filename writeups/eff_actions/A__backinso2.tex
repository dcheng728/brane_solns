\section{Back to SO(2)}
The complete 4-point effective action is already known, but it is commonly written in U(1) covariant form, which makes the 2x2 torus metric a little obscure, so we may try to put it back to SO(2) form.

So here is the strategy:
\begin{itemize}
    \item write everything in the known 4-pt effective action in SO(2) rather than U(1), as in with $F_3, H_4, \tau_1, \tau_2$ rather than $G_3, \mathcal{P}_m, Q_m$; We don't need to use $C, e^{-\Phi}$ because we are already familiar with the metric with $\tau_1, \tau_2$ in SO(2) form.
    \item substitute in metric on the 2x2 torus $M^{ab}$ for $\tau_1, \tau_2$. Note that $M^{11} = \frac{|\tau|^2}{\tau_2}, M^{12} = -\frac{\tau_1}{\tau_2}, M^{22} = \frac{1}{\tau_2}$.
    \item see if there is a 12d interpretation
\end{itemize}

\biggr\}

\subsection{Riemannn tensor}

Given this metric, with the 2x2 torus having indices 10, 11, we compute all nonvanishing components of the Riemann tensor

\begin{equation}
\mathcal{G}_{MN}
= \begin{pmatrix}
    g_{mn} & 0 & 0 \\
    0 & \frac{1}{\tau_2} & \frac{\tau_1}{\tau_2} \\
    0 & \frac{\tau_1}{\tau_2} & \frac{|\tau|^2}{\tau_2}
\end{pmatrix}
\end{equation}

We have that
\begin{equation}
g_{ab} = \frac{1}{\tau_2}\begin{pmatrix}
    1 & \tau_1 \\
    \tau_1 & |\tau|^2
\end{pmatrix}
\end{equation}

From this 12d metric, one immediately finds that the nonvanishing components of the Riemann tensor are $\hat{\Gamma}^p_{mn} = \Gamma^p{}_{mn}$, $\hat{\Gamma}^a{}_{bm} = \frac{1}{2}g^{ac}g_{bc,m}$, $\hat{\Gamma}^m{}_{ab} = - \frac{1}{2}g^{mp}g_{ab,p}$.
And we evaluate that $a$ for rows and $b$ for columns, we have
\begin{equation}
\Gamma^a{}_{bm}
=\frac{1}{2 \tau_2^2}\begin{pmatrix}
    -\tfrac12 \nabla_m|\tau|^2 & |\tau|^2\nabla_m\tau_1 - \tau_1 \nabla_m|\tau|^2 \\
    \nabla_m\tau_1 & \tfrac12 \nabla_m|\tau|^2
\end{pmatrix}
\end{equation}
\begin{equation}
\Gamma_{abm}
= \frac{1}{2\tau_2}\begin{pmatrix}
    -\nabla_m\ln\tau_2
    & \nabla_m\tau_1 - \tau_1\nabla_m\ln\tau_2 \\
    \nabla_m\tau_1 - \tau_1\nabla_m\ln\tau_2 & \nabla_m|\tau|^2 - |\tau|^2\nabla_m\ln\tau_2
\end{pmatrix}
=\frac{1}{2}g_{ab,m}
\end{equation}
\begin{equation}
\Gamma^m{}_{ab} = -\frac{1}{2\tau_2}\begin{pmatrix}
    -\nabla^m\ln\tau_2 & \nabla^m\tau_1 - \tau_1\nabla^m\ln\tau_2 \\
    \nabla^m\tau_1-\tau_1\nabla^m\ln\tau_2 & \nabla^m|\tau|^2 - |\tau|^2\nabla^m\ln\tau_2
\end{pmatrix}
\end{equation}

and we also find
\begin{equation}
\begin{aligned}
&\Gamma^a{}_{cn}\Gamma^c{}_{bm}\\
&=\frac{1}{4\tau_2^4}\begin{pmatrix}
    \tau_2^2\,\nabla_m\tau_1\,\nabla_n\tau_1 + \tau_1\tau_2(\nabla_m\tau_2\,\nabla_n\tau_1 - \nabla_m\tau_1\,\nabla_n\tau_2) + \tau_2^2\,\nabla_m\tau_2\,\nabla_n\tau_2 & |\tau|^2\tau_2(\nabla_m\tau_2\,\nabla_n\tau_1 - \nabla_m\tau_1\,\nabla_n\tau_2) \\
    \tau_2(\nabla_m\tau_1\,\nabla_n\tau_2 - \nabla_m\tau_2\,\nabla_n\tau_1) & \tau_2^2\,\nabla_m\tau_1\,\nabla_n\tau_1 + \tau_1\tau_2(\nabla_m\tau_1\,\nabla_n\tau_2 - \nabla_m\tau_2\,\nabla_n\tau_1) + \tau_2^2\,\nabla_m\tau_2\,\nabla_n\tau_2
\end{pmatrix}
\end{aligned}
\end{equation}
From this we can compute $\Gamma_{acn}\Gamma^{c}{}_{bm} = g_{ad}\Gamma^d{}_{cn}\Gamma^c{}_{bm}$:
\begin{equation}
\begin{aligned}
&\Gamma_{acn}\Gamma^{c}{}_{bm}\\
&=\frac{1}{4\tau_2^3}\begin{pmatrix}
    \nabla_m\tau_1\,\nabla_n\tau_1 + \nabla_m\tau_2\,\nabla_n\tau_2 & \tau_1(\nabla_m\tau_1\,\nabla_n\tau_1 + \nabla_m\tau_2\,\nabla_n\tau_2) + \tau_2(\nabla_m\tau_2\,\nabla_n\tau_1 - \nabla_m\tau_1\,\nabla_n\tau_2) \\
    \tau_1(\nabla_m\tau_1\,\nabla_n\tau_1 + \nabla_m\tau_2\,\nabla_n\tau_2) + \tau_2(\nabla_m\tau_1\,\nabla_n\tau_2 - \nabla_m\tau_2\,\nabla_n\tau_1) & |\tau|^2(\nabla_m\tau_1\,\nabla_n\tau_1 + \nabla_m\tau_2\,\nabla_n\tau_2)
\end{pmatrix}
\end{aligned}
\end{equation}
We can also evaluate $\nabla_n \Gamma^a{}_{bm}$:
\begin{equation}
\begin{aligned}
g_{ad}\nabla_n\Gamma^{d}{}_{bm}
= \;&\frac{1}{2\tau_2^2}\begin{pmatrix}
    -\nabla_n\nabla_m\tau_2 & \tau_2\nabla_n\nabla_m\tau_1 - \tau_1\nabla_n\nabla_m\tau_2 \\
    \tau_2\nabla_n\nabla_m\tau_1 - \tau_1\nabla_n\nabla_m\tau_2 & 2\tau_1\tau_2\nabla_n\nabla_m\tau_1 + (\tau_2^2-\tau_1^2)\nabla_n\nabla_m\tau_2
\end{pmatrix}\\[6pt]
+\;&\frac{1}{2\tau_2^3}\begin{pmatrix}
    -\nabla_m\tau_1\,\nabla_n\tau_1 + \nabla_m\tau_2\,\nabla_n\tau_2 & -\tau_1\nabla_m\tau_1\,\nabla_n\tau_1 - 2\tau_2\nabla_m\tau_2\,\nabla_n\tau_1 + \tau_1\nabla_m\tau_2\,\nabla_n\tau_2 \\
    -\tau_1\nabla_m\tau_1\,\nabla_n\tau_1 - 2\tau_2\nabla_m\tau_1\,\nabla_n\tau_2 + \tau_1\nabla_m\tau_2\,\nabla_n\tau_2 & (\tau_2^2{-}\tau_1^2)\nabla_m\tau_1\,\nabla_n\tau_1 - 2\tau_1\tau_2(\nabla_m\tau_1\,\nabla_n\tau_2{+}\nabla_m\tau_2\,\nabla_n\tau_1) + (\tau_1^2{-}\tau_2^2)\nabla_m\tau_2\,\nabla_n\tau_2
\end{pmatrix}
\end{aligned}
\end{equation}
and we find $R_{ambn} = - \nabla_n\Gamma^a{}_{bm} - \Gamma^a{}_{cn}\Gamma^c{}_{bm}$:

\begin{equation}
\begin{aligned}
R_{10,m,10,n} &= \frac{1}{2\tau_2^2}\nabla_n\nabla_m\tau_2 + \frac{1}{4\tau_2^3}\nabla_m\tau_1\,\nabla_n\tau_1 - \frac{3}{4\tau_2^3}\nabla_m\tau_2\,\nabla_n\tau_2 \\[4pt]

\end{aligned}
\end{equation}
