\begin{equation}
\mathcal{G}_{MN}
= \begin{pmatrix}
    g_{mn} & 0 & 0 \\
    0 & \frac{1}{\tau_2} & \frac{\tau_1}{\tau_2} \\
    0 & \frac{\tau_1}{\tau_2} & \frac{|\tau|^2}{\tau_2}
\end{pmatrix}
\end{equation}

\begin{equation}
(\nabla \mathcal{P})_{mn}{}^{pq} \equiv 
\delta_{[m}^{[p} \nabla_{n]}\mathcal{P}^{q]}
\end{equation}
has the same symmetries as the Riemann tensor.

\begin{align}
\nabla_m\tau_1 &= -2\tau_2 Q_\mu, &
\nabla_m\tau_2 &= -\tau_2 (\mathcal{P}_m + \bar{\mathcal{P}}_m), &
Q_m &= \frac{i}{2}(\mathcal{P}_m - \bar{\mathcal{P}}_m)
\end{align}

\begin{equation}
\mathcal{P}_m = \frac{i}{2\tau_2} \nabla_m \tau, \quad
\bar{\mathcal{P}}_m = -\frac{i}{2\tau_2} \nabla_m \bar{\tau}
\end{equation}

\begin{equation}
D_m \mathcal{P}_n = \nabla_m \mathcal{P}_n - 2i Q_m \mathcal{P}_n,\quad
D_m \bar{\mathcal{P}}_n = \nabla_m \bar{\mathcal{P}}_n + 2i Q_m \bar{\mathcal{P}}_n
\end{equation}
